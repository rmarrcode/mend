import pytest
from sympy import Matrix, MatrixError, eye, symbols, S


class TestMatrixInverse:

    def test_singular_matrix(self):
        """Test that singular matrix (det=0) raises MatrixError"""
        # A singular matrix has determinant = 0
        M = Matrix([[1, 2], [2, 4]])

        # det([[1,2],[2,4]]) = 1*4 - 2*2 = 0 (singular)
        assert M.det() == 0

        # Should raise MatrixError with message about singularity
        with pytest.raises(MatrixError) as exc_info:
            M.inv()

        error_msg = str(exc_info.value).lower()
        assert "singular" in error_msg or "not invertible" in error_msg

    def test_singular_error_message(self):
        """Test that singularity error has clear message"""
        M = Matrix([[0, 0], [0, 0]])

        try:
            result = M.inv()
            # If we get here, test fails - should have raised error
            raise AssertionError("Expected MatrixError for singular matrix but got result: " + str(result))
        except MatrixError as e:
            # Good - got expected error
            error_msg = str(e).lower()
            assert "singular" in error_msg or "not invertible" in error_msg
        except IndexError as e:
            # Bad - got IndexError instead of MatrixError
            raise AssertionError(f"Got IndexError instead of MatrixError: {e}")

    def test_invertible_matrix(self):
        """Test that invertible matrices work correctly"""
        M = Matrix([[1, 2], [3, 4]])

        # det([[1,2],[3,4]]) = 1*4 - 3*2 = -2 (not singular)
        assert M.det() != 0

        # Should return inverse without error
        result = M.inv()

        # Inverse of [[1,2],[3,4]] should be [[-2, 1], [1.5, -0.5]]
        assert result is not None

        # Verify: M * M.inv() = I
        identity = M * result
        assert identity == eye(2)

    def test_identity_matrix(self):
        """Test that identity matrix inv is itself"""
        M = eye(3)
        result = M.inv()
        assert result == M

def eye(n):
    """Create an n x n identity matrix"""
    return Matrix([[1 if i == j else 0 for j in range(n)] for i in range(n)])
